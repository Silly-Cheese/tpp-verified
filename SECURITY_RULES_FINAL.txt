Recommended Final Firestore Rules for The Prayer Project Verification Portal

Use these only after the site has been tested and your admin login is working.

rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    function isAdmin() {
      return request.auth != null && request.auth.token.email == 'pray@ask4prayers.com';
    }

    match /volunteerConfirmations/{documentId} {
      allow read: if true;
      allow create, update, delete: if isAdmin();
    }

    match /admins/{adminId} {
      allow read, write: if isAdmin();
    }

    match /auditLogs/{logId} {
      allow create: if isAdmin();
      allow read: if isAdmin();
      allow update, delete: if false;
    }

    match /system/{docId} {
      allow read: if true;
      allow write: if isAdmin();
    }
  }
}
